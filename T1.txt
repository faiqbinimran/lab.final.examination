import 'dart:async';
import 'dart:math';

Future<int> fetchApiStatus() async {
  await Future.delayed(Duration(seconds: 5));

  List<int> statusCodes = [500, 404, 200];
  return statusCodes[Random().nextInt(statusCodes.length)];
}

void main() async {
  print('Connecting..');

  int statusCode = await fetchApiStatus();

  if (statusCode == 500) {
    print('Internal server error occurred.');
  } else if (statusCode == 404) {
    print('Resource not found.');
  } else if (statusCode == 200) {
    print('Request completed successfully.');
  } else {
    print('Unexpected response received.');
  }
}
